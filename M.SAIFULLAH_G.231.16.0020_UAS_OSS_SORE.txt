def Tampil_Tuple(tuplee):
	for i in range(len(tuplee)):
		print(i+1,'.',tuplee[i])

def Tampil_List(listing):
	for i in listing:
		print(i)

	Tampil_List(listing)

	print("Daftar Nama Dosen FTIK USM ")
